Successfully installed {{ .Release.Name }} {{ .Chart.AppVersion }}.

Check the status by running: kubectl get pod -n {{ .Release.Namespace }}

Common operation and maintenance commands:
  List StatefulSets: kubectl get sts -n {{ .Release.Namespace }}
  Restart StatefulSet: kubectl rollout restart sts -n {{ .Release.Namespace }} <sts>
  Edit StatefulSet: kubectl edit sts -n {{ .Release.Namespace }} <sts>
  Stop StatefulSet: kubectl scale sts -n {{ .Release.Namespace }} --replicas=0 <sts>
  Start StatefulSet: kubectl scale sts -n {{ .Release.Namespace }} --replicas=1 <sts>
  List Pods: kubectl get pods -n {{ .Release.Namespace }}
  Restart Pod: kubectl delete pod -n {{ .Release.Namespace }} <pod>
  Get Pod details: kubectl describe pod -n {{ .Release.Namespace }} <pod>
  Log Pod: kubectl logs -f pod -n {{ .Release.Namespace }} <pod>
  Exec Pod: kubectl exec -it -n {{ .Release.Namespace }} <pod> -- sh
  List ConfigMaps: kubectl get cm -n {{ .Release.Namespace }}
  Edit ConfigMap: kubectl edit cm -n {{ .Release.Namespace }} <configmap>
  Check MongoDB Replicas Status: kubectl exec -it -n {{ .Release.Namespace }} mongodb-0 -- mongo -u{{ .Values.mongodb.rootUser }} -p{{ .Values.mongodb.rootPassword }} --authenticationDatabase admin --eval 'rs.status()'
